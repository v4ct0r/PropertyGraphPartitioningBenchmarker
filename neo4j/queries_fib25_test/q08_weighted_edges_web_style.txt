// Web-derived from neuPrint examples (ConnectsTo-weight style)
MATCH (n:Neuron)-[x]->(m:Neuron)
WHERE toInteger(coalesce(x.weight, 0)) > 10
RETURN n.bodyId AS srcBodyId,
       type(x) AS relType,
       toInteger(coalesce(x.weight, 0)) AS weight,
       m.bodyId AS dstBodyId
ORDER BY weight DESC, srcBodyId ASC, dstBodyId ASC
LIMIT 100
