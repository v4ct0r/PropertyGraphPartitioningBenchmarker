// Web-derived from neuPrint FB/ConnectsTo path examples, adapted to mb6 ROI flags
MATCH (n:Neuron)-[c]->(m:Neuron)
WHERE (n.alpha1 = true OR toString(n.alpha1) = 'true')
  AND (m.alpha1 = true OR toString(m.alpha1) = 'true')
  AND toInteger(coalesce(c.weight, 0)) >= 10
RETURN n.bodyId AS srcBodyId,
       type(c) AS relType,
       toInteger(coalesce(c.weight, 0)) AS weight,
       m.bodyId AS dstBodyId
ORDER BY weight DESC, srcBodyId ASC, dstBodyId ASC
LIMIT 100
